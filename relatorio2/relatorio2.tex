\subsubsection{Parâmetros}
\begin{description}[leftmargin=2cm]
  \item[$c_{ij}$] Custo de \ldots
  \item[$b_i$]    Capacidade de \ldots
  \item[$d_j$]    Demanda de \ldots
\end{description}

\subsubsection{Variáveis de Decisão}
\begin{description}[leftmargin=2cm]
  \item[$x_{ij} \geq 0$] Quantidade de \ldots alocada de $i$ para $j$
\end{description}

\subsubsection{Modelo}
\begin{align}
  \min \quad        & z = \sum_{i \in I} \sum_{j \in J} c_{ij}\, x_{ij}
                    \label{eq:fo_r2} \\
  \text{s.a.} \quad & \sum_{j \in J} x_{ij} \leq b_i,
                    \quad \forall\, i \in I   \label{eq:cap_r2} \\
                    & \sum_{i \in I} x_{ij} \geq d_j,
                    \quad \forall\, j \in J   \label{eq:dem_r2} \\
                    & x_{ij} \geq 0,
                    \quad \forall\, i \in I,\; j \in J   \label{eq:nonneg_r2}
\end{align}

% Descrição das restrições
A função objetivo~\eqref{eq:fo_r2} minimiza o custo total.
As restrições~\eqref{eq:cap_r2} garantem que a capacidade de cada origem não seja excedida.
As restrições~\eqref{eq:dem_r2} asseguram o atendimento de toda a demanda.
As restrições~\eqref{eq:nonneg_r2} impõem a não-negatividade.
